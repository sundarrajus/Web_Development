What is a Promise?

Think of it like ordering food: You place an order (create the promise), get a token (pending), and later either get your food (fulfilled) or hear "sorry, out of stock" (rejected). Meanwhile you don't just stand there — you go sit down (non-blocking). 
// A promise has 3 states: pending → fulfilled or rejected
const p = new Promise((resolve, reject) => {
    let success = true;
    if (success) {
        resolve("Data loaded!");      // fulfilled
    } else {
        reject("Something failed.");  // rejected
    }
});

.then() & .catch() chaining

function loadUser() {
    return new Promise((resolve, reject) => {
        setTimeout(() => resolve({ name: "Sundar", age: 21 }), 1000);
    });
}

function loadPosts(user) {
    return new Promise((resolve) => {
        setTimeout(() => resolve([`${user.name}'s post 1`, `${user.name}'s post 2`]), 800);
    });
}

// Chaining — each .then() receives the resolved value of the previous
loadUser()
    .then(user => {
        console.log("User:", user.name);
        return loadPosts(user);          // return next promise
    })
    .then(posts => {
        console.log("Posts:", posts);
    })
    .catch(err => {
        console.error("Error:", err);    // handles any rejection in the chain
    })
    .finally(() => {
        console.log("Done (always runs)");
    });

Promise.all() & Promise.race()

const p1 = new Promise(res => setTimeout(() => res("Result 1"), 1000));
const p2 = new Promise(res => setTimeout(() => res("Result 2"), 500));
const p3 = new Promise(res => setTimeout(() => res("Result 3"), 800));

// Promise.all — waits for ALL to finish; fails if ANY one rejects
Promise.all([p1, p2, p3]).then(results => {
    console.log(results);   // ["Result 1", "Result 2", "Result 3"]
});

// Promise.race — resolves/rejects with the FIRST to settle
Promise.race([p1, p2, p3]).then(winner => {
    console.log(winner);    // "Result 2"  (fastest, 500ms)
});

async / await

async/await is syntactic sugar over Promises — it makes async code look and read like synchronous code. Under the hood it still uses Promises. 
// async keyword makes a function return a Promise
async function fetchData() {
    // await pauses execution until the Promise resolves
    const user  = await loadUser();
    const posts = await loadPosts(user);
    console.log(user.name, posts);
}
fetchData();

try/catch with async/await

function riskyOperation() {
    return new Promise((resolve, reject) => {
        setTimeout(() => reject("Network timeout!"), 1000);
    });
}

async function run() {
    try {
        const result = await riskyOperation();
        console.log("Success:", result);
    } catch (err) {
        console.error("Caught error:", err);   // "Network timeout!"
    } finally {
        console.log("Cleanup done.");
    }
}
run();

Callback vs Promise vs async/await — same task

// ---------- CALLBACK (messy) ----------
getUser(1, (user) => {
    getPosts(user.id, (posts) => {
        getComments(posts[0].id, (comments) => {
            console.log(comments);
        });
    });
});

// ---------- PROMISE CHAIN (cleaner) ----------
getUser(1)
    .then(user => getPosts(user.id))
    .then(posts => getComments(posts[0].id))
    .then(comments => console.log(comments))
    .catch(err => console.error(err));

// ---------- ASYNC/AWAIT (cleanest) ----------
async function showComments() {
    try {
        const user     = await getUser(1);
        const posts    = await getPosts(user.id);
        const comments = await getComments(posts[0].id);
        console.log(comments);
    } catch (err) {
        console.error(err);
    }
}
showComments();
