What is a Callback?

Definition: A callback is a function passed as an argument to another function, to be called later — either immediately or after something finishes (like a timer or network request). 
// A function that accepts another function as argument
function doMath(a, b, operation) {
    return operation(a, b);
}

function add(x, y)      { return x + y; }
function multiply(x, y) { return x * y; }

console.log(doMath(3, 4, add));       // 7
console.log(doMath(3, 4, multiply));  // 12

// Using an arrow function inline as callback
console.log(doMath(10, 5, (x, y) => x - y));   // 5

forEach — iterate without return

const students = ["Sundar", "Raju", "Priya"];

// forEach runs a callback for each element (returns undefined)
students.forEach((student, index) => {
    console.log(`${index + 1}. ${student}`);
});
// 1. Sundar
// 2. Raju
// 3. Priya

map — transform each element

const marks = [45, 78, 92, 61];
const grades = marks.map(m => {
    if (m >= 90) return "A";
    if (m >= 75) return "B";
    if (m >= 60) return "C";
    return "D";
});
console.log(grades);   // ["D", "B", "A", "C"]

filter — keep matching elements

const scores = [32, 78, 55, 90, 41, 88];
const passed = scores.filter(s => s >= 60);
console.log(passed);   // [78, 90, 88]

find / every / some

const nums = [3, 7, 2, 9, 4];

// find — returns FIRST element that matches (or undefined)
console.log(nums.find(n => n > 5));    // 7

// every — returns true only if ALL elements match
console.log(nums.every(n => n > 0));   // true
console.log(nums.every(n => n > 5));   // false

// some — returns true if AT LEAST ONE element matches
console.log(nums.some(n => n > 8));    // true
console.log(nums.some(n => n > 10));   // false

sort — with a comparator callback

Note: Without a callback, sort() sorts lexicographically (string order), so [10, 9, 2] becomes [10, 2, 9]. Always pass a comparator for numbers. 
const prices = [200, 45, 1500, 30, 890];

// Ascending (a - b gives positive when a > b → move a right)
prices.sort((a, b) => a - b);
console.log(prices);   // [30, 45, 200, 890, 1500]

// Descending
prices.sort((a, b) => b - a);
console.log(prices);   // [1500, 890, 200, 45, 30]

// Sort objects by a property
const students = [
    { name: "Raju", marks: 72 },
    { name: "Priya", marks: 91 },
    { name: "Sundar", marks: 85 },
];
students.sort((a, b) => b.marks - a.marks);
console.log(students[0].name);   // "Priya"  (highest marks first)

setTimeout & setInterval

// setTimeout — run callback ONCE after a delay (milliseconds)
setTimeout(() => {
    console.log("Runs after 2 seconds");
}, 2000);

// setInterval — run callback REPEATEDLY at an interval
const id = setInterval(() => {
    console.log("Runs every 1 second");
}, 1000);

// clearInterval — stop the interval
clearInterval(id);

Callback Hell — why it becomes a problem

Problem: When you need multiple async operations in sequence, callbacks nest deeply — making code hard to read and maintain. This is called "callback hell" or the "pyramid of doom." Promises and async/await (next topic) solve this. 
// Simulating sequential async tasks with callbacks (messy!)
setTimeout(() => {
    console.log("Step 1: Login");
    setTimeout(() => {
        console.log("Step 2: Load profile");
        setTimeout(() => {
            console.log("Step 3: Load posts");
            setTimeout(() => {
                console.log("Step 4: Done");
            }, 1000);
        }, 1000);
    }, 1000);
}, 1000);

// This is callback hell — hard to read, hard to handle errors.
// Solution → Promises & async/await (see next file).
