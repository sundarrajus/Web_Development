Class Syntax & Constructor

Class is a blueprint. Instance is the real object created from the blueprint using new. 
class Student {
    // constructor runs automatically when 'new Student()' is called
    constructor(name, age) {
        this.name = name;   // 'this' refers to the new instance being created
        this.age  = age;
        this.marks = [];
    }

    // instance method — available on every Student object
    greet() {
        console.log(`Hi, I am ${this.name}, ${this.age} years old.`);
    }

    addMark(mark) {
        this.marks.push(mark);
    }

    average() {
        if (this.marks.length === 0) return 0;
        return this.marks.reduce((a, b) => a + b, 0) / this.marks.length;
    }
}

const s1 = new Student("Sundar", 21);
s1.greet();             // Hi, I am Sundar, 21 years old.
s1.addMark(85);
s1.addMark(92);
console.log(s1.average());   // 88.5

Getters & Setters

class Circle {
    constructor(radius) {
        this._radius = radius;   // underscore = convention for "private-ish"
    }

    get radius() { return this._radius; }

    set radius(value) {
        if (value <= 0) throw new RangeError("Radius must be positive");
        this._radius = value;
    }

    get area() {
        return (Math.PI * this._radius ** 2).toFixed(2);
    }
}

const c = new Circle(5);
console.log(c.area);     // "78.54"   (called like a property, not a method)
c.radius = 10;
console.log(c.area);     // "314.16"
// c.radius = -1;        // RangeError: Radius must be positive

Static Methods & Properties

class MathUtils {
    static PI = 3.14159;   // static property — belongs to the CLASS, not instances

    static square(n)   { return n * n; }
    static cube(n)     { return n * n * n; }
    static isPrime(n) {
        if (n < 2) return false;
        for (let i = 2; i <= Math.sqrt(n); i++) {
            if (n % i === 0) return false;
        }
        return true;
    }
}

// Called on the class directly — NOT on an instance
console.log(MathUtils.PI);          // 3.14159
console.log(MathUtils.square(4));   // 16
console.log(MathUtils.isPrime(17)); // true

Inheritance with extends & super

class Animal {
    constructor(name, sound) {
        this.name  = name;
        this.sound = sound;
    }

    speak() {
        console.log(`${this.name} says ${this.sound}!`);
    }

    toString() {
        return `Animal(${this.name})`;
    }
}

class Dog extends Animal {
    constructor(name, breed) {
        super(name, "Woof");   // super() calls Animal's constructor
        this.breed = breed;
    }

    // Method overriding — replaces Animal's speak()
    speak() {
        super.speak();         // optionally call parent method
        console.log(`${this.name} wags its tail.`);
    }

    fetch(item) {
        console.log(`${this.name} fetches the ${item}!`);
    }
}

const dog = new Dog("Bruno", "Labrador");
dog.speak();         // Bruno says Woof!  /  Bruno wags its tail.
dog.fetch("ball");   // Bruno fetches the ball!

console.log(dog instanceof Dog);    // true
console.log(dog instanceof Animal); // true — Dog IS-A Animal
