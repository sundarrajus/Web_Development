How It Works

getElementById

// Select element by ID, change text and style
let h = document.getElementById("heading-demo");
h.innerText = 'Changed by JS';
h.style.color = 'red';

getElementsByClassName

Note: Returns an HTMLCollection (array-like). Loop through it to access each element. 
let items = document.getElementsByClassName("hi");
for (let i = 0; i < items.length; i++) {
    items[i].innerText = 'sundar raju';
    items[i].style.color = 'blue';
}
items[1].style.color = 'green';   // override one specifically

getElementsByTagName

let headings = document.getElementsByTagName("h3");
headings[0].style.color = 'orange';

querySelector & querySelectorAll

// querySelector — selects the FIRST matching element
let first = document.querySelector(".question");
first.innerText = "First Selected";
first.style.color = 'green';

let byId = document.querySelector("#q2");
byId.innerText = 'Selected by #id';

// querySelectorAll — selects ALL matching elements (NodeList)
let allQ = document.querySelectorAll(".question");
for (let i = 0; i < allQ.length; i++) {
    allQ[i].style.color = "purple";
    allQ[i].innerText = "Question " + (i + 1);
}

innerHTML

// innerHTML parses HTML tags; innerText treats content as plain text
let div = document.getElementById("container");
div.innerHTML = "<strong>Bold content injected!</strong>";

Reading Input Values

function showName() {
    let val = document.getElementById("nameInput").value;
    document.getElementById("nameOutput").innerText = "Hello, " + val;
}

classList.add()

let p = document.getElementById("classDemo");
p.classList.add("two");   // adds class 'two', applies its CSS rules

// CSS for .two:
// .two { color: red; font-weight: bold; }
