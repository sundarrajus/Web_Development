Common Error Types

// SyntaxError — invalid code (caught before runtime)
// let x = ;   ← SyntaxError: Unexpected token

// ReferenceError — using an undeclared variable
console.log(y);   // ReferenceError: y is not defined

// TypeError — wrong type for an operation
null.toString();  // TypeError: Cannot read properties of null

// RangeError — value out of allowed range
new Array(-1);    // RangeError: Invalid array length

try / catch / finally

function divide(a, b) {
    if (b === 0) throw new Error("Cannot divide by zero!");
    return a / b;
}

try {
    let result = divide(10, 0);    // throws an error
    console.log(result);           // this line is skipped
} catch (err) {
    // err is the Error object — has .name, .message, .stack
    console.log(err.name);         // "Error"
    console.log(err.message);      // "Cannot divide by zero!"
} finally {
    // always runs — use for cleanup (close DB, hide spinner, etc.)
    console.log("Calculation attempted.");
}

The Error Object

try {
    undefined.toString();
} catch (err) {
    console.log(err.name);     // "TypeError"
    console.log(err.message);  // "Cannot read properties of undefined"
    console.log(err.stack);    // Full stack trace (useful for debugging)
}

throw — custom errors

function validateAge(age) {
    if (typeof age !== "number") throw new TypeError("Age must be a number");
    if (age < 0)   throw new RangeError("Age cannot be negative");
    if (age > 150) throw new RangeError("Age seems unrealistic");
    return true;
}

try {
    validateAge("twenty");
} catch (err) {
    console.log(err.name + ": " + err.message);   // TypeError: Age must be a number
}

try {
    validateAge(-5);
} catch (err) {
    console.log(err.name + ": " + err.message);   // RangeError: Age cannot be negative
}

Custom Error Classes

class ValidationError extends Error {
    constructor(message, field) {
        super(message);
        this.name = "ValidationError";
        this.field = field;
    }
}

class NetworkError extends Error {
    constructor(message, statusCode) {
        super(message);
        this.name = "NetworkError";
        this.statusCode = statusCode;
    }
}

try {
    throw new ValidationError("Email is required", "email");
} catch (err) {
    if (err instanceof ValidationError) {
        console.log(`Field "${err.field}": ${err.message}`);
    } else {
        console.error("Unexpected error:", err);
    }
}

Error Handling in async/await

async function fetchData(url) {
    try {
        const res = await fetch(url);
        if (!res.ok) throw new NetworkError("Request failed", res.status);
        const data = await res.json();
        return data;
    } catch (err) {
        if (err instanceof NetworkError) {
            console.error(`Network issue (${err.statusCode}): ${err.message}`);
        } else {
            console.error("Unexpected:", err.message);
        }
        return null;
    }
}

Best Practices

Do: Always use try/catch around code that can fail (user input, network, JSON parsing). Throw meaningful error messages. Use finally for cleanup. Distinguish error types with instanceof.

Don't: Use empty catch blocks (catch (e) {}) — they silently swallow errors and make debugging very hard. 
// BAD — silently swallows error
try {
    riskyOperation();
} catch (e) {}   // never do this

// GOOD — at minimum, log it
try {
    riskyOperation();
} catch (e) {
    console.error("riskyOperation failed:", e.message);
}
