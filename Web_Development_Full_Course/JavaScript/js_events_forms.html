How It Works

addEventListener

Note: addEventListener is preferred over inline onclick attributes — it keeps HTML and JS separate and allows multiple listeners on the same element. 
let btn = document.getElementById("btn1");

// 'dblclick' fires when the button is double-clicked
btn.addEventListener("dblclick", () => {
    console.log("Button double-clicked");
    btn.style.backgroundColor = "red";
});

Keyboard Events: keydown & keyup

let input = document.getElementById("keyInput");

// keydown — fires as soon as a key is pressed
input.addEventListener("keydown", () => {
    console.log("Key pressed down");
});

// keyup — fires when the key is released
input.addEventListener("keyup", () => {
    console.log("Key released");
});

Form Events

const form  = document.getElementById("myForm");
const name  = document.getElementById("nameField");
const msg   = document.getElementById("formMsg");

// submit — fires when the form is submitted
// e.preventDefault() stops the page from reloading
form.addEventListener("submit", (e) => {
    e.preventDefault();
    if (name.value === "") {
        msg.innerText = "Name is required";
        msg.style.color = "red";
    } else {
        msg.innerText = "Form submitted successfully!";
        msg.style.color = "green";
    }
});

// focus — fires when the input gains focus (user clicks into it)
name.addEventListener("focus", () => {
    name.style.border = "2px solid red";
});

// blur — fires when the input loses focus (user clicks away)
name.addEventListener("blur", () => {
    if (name.value === "") {
        msg.innerText = "Name cannot be empty";
    }
});

// input — fires on every keystroke while typing
name.addEventListener("input", () => {
    msg.innerText = "Typing...";
});

// change — fires when the value is committed (after blur with a changed value)
name.addEventListener("change", () => {
    console.log("Value changed to:", name.value);
});
