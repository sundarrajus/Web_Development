What is JSON?

JSON (JavaScript Object Notation) is a text format for storing and sending data. APIs send data as JSON strings; we parse them into JS objects to use them. 
// JS object → JSON string
const user = { name: "Sundar", age: 21, skills: ["JS", "CSS"] };
const jsonStr = JSON.stringify(user);
console.log(jsonStr);
// '{"name":"Sundar","age":21,"skills":["JS","CSS"]}'
console.log(typeof jsonStr);   // "string"

// JSON string → JS object
const parsed = JSON.parse(jsonStr);
console.log(parsed.name);      // "Sundar"
console.log(typeof parsed);    // "object"

// Pretty-print with indentation
console.log(JSON.stringify(user, null, 2));

fetch() — GET request

// fetch() returns a Promise that resolves to a Response object
fetch("https://jsonplaceholder.typicode.com/users/1")
    .then(response => {
        // response.ok is true for 200-299 status codes
        if (!response.ok) {
            throw new Error("HTTP error! status: " + response.status);
        }
        return response.json();    // parse JSON body — also returns a Promise
    })
    .then(user => {
        console.log(user.name);    // "Leanne Graham"
        console.log(user.email);   // "Sincere@april.biz"
    })
    .catch(err => {
        console.error("Fetch failed:", err);
    });

fetch() with async/await (cleaner)

async function getUser(id) {
    try {
        const response = await fetch(`https://jsonplaceholder.typicode.com/users/${id}`);

        if (!response.ok) {
            throw new Error(`HTTP ${response.status}`);
        }

        const user = await response.json();
        console.log(user.name, user.email);
        return user;
    } catch (err) {
        console.error("Error:", err.message);
    }
}

getUser(1);

POST request — sending data

async function createPost() {
    const newPost = {
        title: "My first post",
        body:  "Learning Fetch API is fun!",
        userId: 1
    };

    const response = await fetch("https://jsonplaceholder.typicode.com/posts", {
        method:  "POST",
        headers: { "Content-Type": "application/json" },
        body:    JSON.stringify(newPost)   // must stringify the object
    });

    const data = await response.json();
    console.log(data);          // { id: 101, title: "...", body: "...", userId: 1 }
    console.log(response.status); // 201 (Created)
}

createPost();

Common HTTP Methods

// GET    — read data          fetch(url)
// POST   — create new data    fetch(url, { method: "POST",   body: ... })
// PUT    — replace data       fetch(url, { method: "PUT",    body: ... })
// PATCH  — update part        fetch(url, { method: "PATCH",  body: ... })
// DELETE — delete data        fetch(url, { method: "DELETE" })
