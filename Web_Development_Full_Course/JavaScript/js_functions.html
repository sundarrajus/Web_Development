Functions & Scope

Function Declaration

Hoisting: Function declarations are hoisted — you can call them before they appear in the code. 
// Can call before definition due to hoisting
fun();
function fun() {
    console.log('hello');
}

// Calling a function and using its return value
let n = f();
console.log(n * 20);    // 200

function f() {
    let n1 = 2;
    let n2 = 8;
    let r = n1 + n2;
    return r;           // returns 10
}

Function Expression

// Named function expression stored in a variable
let a = function add(a, b) {
    console.log(a * b);
};
a(4, 8);    // 32

Arrow Function

let c = (a, b) => {
    console.log(a * b);
};
c(4, 6);    // 24

arguments Object

Note: The arguments object is available inside regular functions (not arrow functions) and holds all passed values, regardless of how many parameters are defined. 
function s() {
    for (let i = 0; i < arguments.length; i++) {
        console.log("hello");
    }
}
s(3);   // prints "hello" once
s();    // prints nothing (no arguments)

Primitive vs Reference Types

// Primitive — copied by VALUE
let a = 10;
let b = a;
a = 20;
console.log(a, b);    // 20  10  (b is not affected)

// Object — copied by REFERENCE
let a = { no: 2 };
let b = a;
a.no = 90;
console.log(a.no, b.no);    // 90  90  (both point to same object)

// Array — also reference type
function main() {
    let l = [1, 2, 3, 4, 5];
    f(l);
    for (let i = 0; i < l.length; i++) {
        console.log(l[i]);    // 1 2 3 45 5  (l[3] changed inside f)
    }
}
function f(a) {
    a[3] = 45;    // modifies the original array
}
main();
