Modern JS Syntax (ES6+)

let vs const vs var

Rule of thumb: Always use const by default. Switch to let only when you know the variable will be reassigned. Avoid var — it has function scope and hoisting quirks. 
const PI = 3.14159;       // cannot be reassigned
let count = 0;            // can be reassigned
count = 1;                // OK

const arr = [1, 2, 3];
arr.push(4);              // OK — const protects the binding, not the contents
// arr = [5, 6];          // ERROR — cannot reassign the variable itself

Array Destructuring

const colors = ["red", "green", "blue"];

// Old way
let r = colors[0];
let g = colors[1];

// Destructuring
const [red, green, blue] = colors;
console.log(red);    // "red"
console.log(green);  // "green"

// Skip elements with a comma
const [first, , third] = colors;
console.log(first, third);   // "red" "blue"

// Swap two variables
let a = 1, b = 2;
[a, b] = [b, a];
console.log(a, b);   // 2 1

Object Destructuring

const user = { name: "Sundar", age: 21, city: "Chennai" };

// Old way
let name1 = user.name;

// Destructuring
const { name, age, city } = user;
console.log(name, age);   // "Sundar" 21

// Rename while destructuring
const { name: studentName } = user;
console.log(studentName);   // "Sundar"

// Default values
const { course = "JavaScript" } = user;
console.log(course);   // "JavaScript"  (not in user, so default applies)

// In function parameters
function greet({ name, age }) {
    console.log(`Hello ${name}, you are ${age}`);
}
greet(user);   // Hello Sundar, you are 21

Spread Operator (...)

// Spread in arrays — expand elements
const a = [1, 2, 3];
const b = [4, 5, 6];
const combined = [...a, ...b];
console.log(combined);   // [1, 2, 3, 4, 5, 6]

// Copy an array (not by reference)
const copy = [...a];
copy.push(99);
console.log(a);     // [1, 2, 3]   (original unchanged)
console.log(copy);  // [1, 2, 3, 99]

// Spread in objects — merge
const defaults = { theme: "light", lang: "en" };
const userPrefs = { lang: "ta", font: "large" };
const merged = { ...defaults, ...userPrefs };
console.log(merged);
// { theme: "light", lang: "ta", font: "large" }
// Note: userPrefs.lang overrides defaults.lang

// Spread to pass array as function arguments
const nums = [10, 20, 30];
console.log(Math.max(...nums));   // 30

Rest Parameters (...)

Same syntax, different purpose: Spread expands an iterable. Rest collects remaining arguments into an array. 
// Rest collects all extra arguments
function sum(...numbers) {
    return numbers.reduce((total, n) => total + n, 0);
}
console.log(sum(1, 2, 3));         // 6
console.log(sum(10, 20, 30, 40));  // 100

// Rest after named params
function intro(name, age, ...hobbies) {
    console.log(`${name} (${age}) likes: ${hobbies.join(", ")}`);
}
intro("Sundar", 21, "coding", "chess", "music");
// Sundar (21) likes: coding, chess, music

Default Parameters

function greet(name = "Guest", greeting = "Hello") {
    console.log(`${greeting}, ${name}!`);
}
greet();                    // Hello, Guest!
greet("Sundar");            // Hello, Sundar!
greet("Sundar", "Hi");      // Hi, Sundar!

Short-circuit, Optional Chaining & Nullish Coalescing

// && — returns first falsy value, or last value if all truthy
console.log(0 && "hello");    // 0
console.log(1 && "hello");    // "hello"

// || — returns first truthy value
console.log(0 || "default");  // "default"
console.log("hi" || "nope");  // "hi"

// ?. — optional chaining (safe access, no error if undefined)
const user = { profile: { avatar: "pic.jpg" } };
console.log(user?.profile?.avatar);   // "pic.jpg"
console.log(user?.address?.city);     // undefined  (no error thrown)

// ?? — nullish coalescing (only uses fallback for null/undefined)
let score = 0;
console.log(score || 10);    // 10  (wrong — 0 is falsy)
console.log(score ?? 10);    // 0   (correct — 0 is not null/undefined)

Shorthand Object Properties

const name = "Sundar";
const age = 21;

// Old way
const user1 = { name: name, age: age };

// Shorthand (ES6) — when key and variable name are the same
const user2 = { name, age };
console.log(user2);   // { name: "Sundar", age: 21 }

// Shorthand methods
const obj = {
    greet() {
        console.log("hello");
    }
};
obj.greet();   // hello
