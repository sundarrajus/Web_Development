Objects & Arrays

Objects

// Object literal
let user = {
    uid: 101,
    usn: "sundar",
    age: 21,
};
console.log(user);          // { uid: 101, usn: 'sundar', age: 21 }
console.log(user.age);      // 21  (dot notation)
console.log(user['uid']);   // 101 (bracket notation)

// Dynamic property access
let userinfo = "usn";
console.log(user.userinfo);     // undefined (looks for key 'userinfo')
console.log(user["userinfo"]);  // undefined (same)
console.log(user[userinfo]);    // 'sundar'  (uses variable value 'usn')

Objects with Methods

let s = {
    name: "sundar",
    age: 20,
    course: "python",
    add: function() { console.log("hi I am sundar"); }
};

console.log(s.age);     // 20
console.log(s["name"]); // sundar
console.log(s.add());   // hi I am sundar

Arrays

// Two ways to create arrays
let arr = [1, 2, 3];            // array literal
let arr2 = new Array(1, 2, 3); // array constructor

console.log(Array.isArray(arr));   // true
console.log(Array.isArray({}));    // false (object, not array)

// Modifying elements
arr = ["hi", 123, "hello"];
arr[1] = "sundar";
console.log(arr);       // ["hi", "sundar", "hello"]
console.log(arr.length); // 3

for...in Loop (iterates indices/keys)

Note: for...in on an array gives you the index (0, 1, 2…), not the value. Use for...of or a regular for loop to get values. 
let arr2 = [10, "web", true, null, undefined];
for (let ele in arr2) {
    console.log(ele);   // 0, 1, 2, 3, 4  (indices, not values)
}

Higher-Order Array Methods

let arr = [1, 2, 3];

// map — transforms each element, returns new array
let doubled = arr.map(n => n * 2);
console.log(doubled);   // [2, 4, 6]

// filter — returns elements that pass the test
let evens = arr.filter(n => n % 2 == 0);
console.log(evens);     // [2]

// reduce — reduces to a single value (here: product)
let product = arr.reduce((a, b) => a * b, 1);
console.log(product);   // 6

Mutating Array Methods

let ar = [1, 2, 3, 4, 5];

// push — adds to end, returns new length
let a1 = ar.push(6, 7, 80);
console.log(ar);    // [1,2,3,4,5,6,7,80]
console.log(a1);    // 8  (new length)

// pop — removes last element, returns it
let a2 = ar.pop();
console.log(ar);    // [1,2,3,4,5,6,7]
console.log(a2);    // 80

// shift — removes first element, returns it
let a3 = ar.shift();
console.log(ar);    // [2,3,4,5,6,7]
console.log(a3);    // 1

// unshift — adds to beginning, returns new length
let a4 = ar.unshift(90);
console.log(ar);    // [90,2,3,4,5,6,7]
console.log(a4);    // 7

// splice(start, deleteCount) — removes elements
let a5 = ar.splice(2, 2);
console.log(ar);    // [90,2,5,6,7]
console.log(a5);    // [3,4]  (removed)

// splice(start, 0, ...items) — inserts without removing
let a6 = ar.splice(2, 0, 7, 45, 12);
console.log(ar);    // [90,2,7,45,12,5,6,7]
console.log(a6);    // []  (nothing removed)
