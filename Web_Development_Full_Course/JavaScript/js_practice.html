Prime Number Checker

Prime Number Check

Logic: A prime number has exactly 2 factors — 1 and itself. Count all divisors; if the count is 2, the number is prime. 
let a = 17;   // number to check
let c = 0;    // factor counter

for (let i = 1; i <= a; i++) {
    if (a % i == 0) {
        c += 1;   // i is a factor
    }
}

if (c == 2) {
    console.log(`${a} is a prime number`);
} else {
    console.log(`${a} is NOT a prime number`);
}
