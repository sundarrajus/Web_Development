Creating Strings

let s1 = 'single quotes';
let s2 = "double quotes";
let name = "Sundar";
let s3 = `template literal — hello ${name}!`;   // hello Sundar!

console.log(s3.length);   // 28

toUpperCase & toLowerCase

let str = "Hello World";
console.log(str.toUpperCase());   // HELLO WORLD
console.log(str.toLowerCase());   // hello world

trim / trimStart / trimEnd

Use case: Always trim() user input before processing to remove accidental spaces. 
let s = "   hello   ";
console.log(s.trim());          // "hello"
console.log(s.trimStart());     // "hello   "
console.log(s.trimEnd());       // "   hello"

includes / startsWith / endsWith / indexOf

let str = "JavaScript is awesome";
console.log(str.includes("awesome"));     // true
console.log(str.startsWith("Java"));      // true
console.log(str.endsWith("awesome"));     // true
console.log(str.indexOf("is"));           // 11  (first match position)
console.log(str.lastIndexOf("a"));        // 18  (last match position)
console.log(str.indexOf("python"));       // -1  (not found)

slice & substring

let str = "JavaScript";
//         0123456789

console.log(str.slice(0, 4));       // "Java"
console.log(str.slice(4));          // "Script"
console.log(str.slice(-6));         // "Script"  (negative = from end)

console.log(str.substring(0, 4));   // "Java"  (no negative indices)

replace & replaceAll

let str = "I love cats. Cats are cute.";
console.log(str.replace("cats", "dogs"));      // "I love dogs. Cats are cute."  (first only)
console.log(str.replaceAll("cats", "dogs"));   // "I love dogs. Cats are cute."  (case-sensitive)

// Use regex for case-insensitive replace-all
console.log(str.replace(/cats/gi, "dogs"));    // "I love dogs. dogs are cute."

split & join

let csv = "HTML,CSS,JavaScript,React";
let arr = csv.split(",");
console.log(arr);               // ["HTML", "CSS", "JavaScript", "React"]
console.log(arr.join(" | "));   // "HTML | CSS | JavaScript | React"
console.log(arr.join(""));      // "HTMLCSSJavaScriptReact"

let sentence = "hello world";
console.log(sentence.split(""));   // ["h","e","l","l","o"," ","w","o","r","l","d"]

repeat / padStart / padEnd / charAt

console.log("ha".repeat(3));            // "hahaha"
console.log("5".padStart(3, "0"));      // "005"  (useful for formatting IDs)
console.log("5".padEnd(3, "0"));        // "500"

let str = "Hello";
console.log(str.charAt(1));             // "e"
console.log(str.charCodeAt(0));         // 72  (ASCII code of 'H')
